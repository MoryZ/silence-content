### 7️⃣ Resource 层

#### CodeGenModuleResource (现有框架参考)
```java
// 文件: CodeGenModuleResource.java
// 位置: api/

@Validated
@RestController
@RequestMapping("/api/v1")
public class CodeGenModuleResource implements CodeGenModuleService {
    
    private final CodeGenModuleRepository codeGenModuleRepository;
    private final CodeGenModuleMapper codeGenModuleMapper;
    
    public CodeGenModuleResource(CodeGenModuleRepository codeGenModuleRepository,
                                CodeGenModuleMapper codeGenModuleMapper) {
        this.codeGenModuleRepository = codeGenModuleRepository;
        this.codeGenModuleMapper = codeGenModuleMapper;
    }
    
    @GetMapping("/codeGenModules/{id}")
    public CodeGenModuleView findById(@PathVariable BigInteger id) {
        return codeGenModuleRepository.findById(id, CodeGenModuleView.class)
                .orElseThrow(ResourceNotFoundException::new);
    }
    
    @GetMapping(value = "/codeGenModules", params = {"pageNo", "pageSize"})
    public Page<CodeGenModuleView> queryPage(CodeGenModuleQuery query, Pageable pageable) {
        var criteria = QueryCriteriaConverter.convert(query, CodeGenModule.class);
        return codeGenModuleRepository.findByCriteria(criteria, pageable, CodeGenModuleView.class);
    }
    
    @PostMapping("/codeGenModules")
    public BigInteger create(@RequestBody CodeGenModuleCommand command) {
        var codeGenModule = codeGenModuleMapper.convert(command);
        codeGenModuleRepository.create(codeGenModule);
        return codeGenModule.getId();
    }
    
    @PutMapping("/codeGenModules/{id}")
    public void update(@PathVariable BigInteger id, @RequestBody CodeGenModuleCommand command) {
        var codeGenModule = codeGenModuleMapper.convert(command);
        codeGenModule.setId(id);
        validateModifyingResult(codeGenModuleRepository.update(codeGenModule));
    }
    
    @DeleteMapping("/codeGenModules/{id}")
    public void deleteById(@PathVariable BigInteger id) {
        validateModifyingResult(codeGenModuleRepository.deleteById(id));
    }
}
```
